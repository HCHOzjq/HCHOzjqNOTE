#include<iostream>
#include<cstring>
#include<cstdio>
using namespace std;
const int maxn=100005; 
struct edge
{
	int v;
	long long val;int nxt;
}e[maxn*2];
int cnt;
int head[maxn];
void add_edge(int u,int v,long long val)
{
	e[cnt].v=v,e[cnt].val=val;
	e[cnt].nxt=head[u];head[u]=cnt++;
}
long long point[maxn];

struct Tire01
{
	int ch[100005*32][2];
	long long num[100005*32];int cntt;
	void init()
	{
		ch[0][1]=ch[0][0]=0;cntt=1;
	}
	void insert(long long x)
	{
		int root=0;
		
		for(int i=31;i>=0;i--)
		{
			int y=(x>>i)&1;
			if(!ch[root][y]){
				ch[cntt][0]=ch[cntt][1]=0;
				num[cntt]=0;
				ch[root][y]=cntt++;
			}
			root=ch[root][y];
		}
		num[root]=x;
	}
	long long find(long long x)
	{
		int root=0;
		
		for(int i=31;i>=0;i--)
		{
			int y=(x>>i)&1;
			if(ch[root][y^1])
			root=ch[root][y^1];
			else root=ch[root][y];
		}
		return num[root]^x;
	}
}T;

void dfs(int u,int fa,long long val)
{
	point[u]=val;
	T.insert(val);
	for(int i=head[u];~i;i=e[i].nxt){
		int v=e[i].v;
		if(v!=fa){
			dfs(v,u,val^e[i].val);
		}
	}
}

int main()
{
	int n;
	while(~scanf("%d",&n))
	{
		memset(head,-1,sizeof head);
		cnt=0;
		T.init();
		for(int i=0;i<n-1;i++)
		{
			int u,v;long long val;
			scanf("%d%d%lld",&u,&v,&val);
			add_edge(u,v,val);
			add_edge(v,u,val);
		}
		dfs(0,-1,0);
		long long ans=0;
		for(int i=0;i<n;i++){
			ans=max(ans,T.find(point[i]));
		}
		printf("%lld\n",ans);
	} 
}
