### 欧拉回路

欧拉回路，无向图：必须无奇度点。然后dfs求得。

要反向输出。

```
#include<bits/stdc++.h>

using namespace std;
const int maxn=100;
int tu[maxn][maxn];
map<int,int> mp;
void dfs(int v)
{
	for(int i=1;i<=50;i++)
	{
		if(tu[v][i])
		{
			tu[v][i]--;
			tu[i][v]--;
			dfs(i);
			cout<<i<<" "<<v<<endl;//反向输出,重点
		}
	}
}

int main()
{
	int t;
	cin>>t;
	for(int i=1;i<=t;i++)
	{
		if(i!=1)cout<<endl;
		int n;cin>>n;
		memset(tu,0,sizeof tu);
		mp.clear();
		for(int i=1;i<=n;i++)
		{
			int x,y;
			cin>>x>>y;
			tu[x][y]++;
			tu[y][x]++;
			mp[x]++;
			mp[y]++;
		}
		int flag=1;
		int v=0;
		int maxv=0;
		for(auto it:mp)
		{
			if(it.second%2){
				flag=0;break;
			}
			else if(it.second>maxv){
				maxv=it.second;
				v=it.first;
			}
		}
		cout<<"Case #"<<i<<endl;
		if(flag){
			dfs(v);
		}else {
			cout<<"some beads may be lost"<<endl;
		}
	}
} 
```
