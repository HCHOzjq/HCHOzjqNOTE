#include<bits/stdc++.h>

using namespace std;


using namespace std;
const int maxn=200010;


struct edge
{
	int v,nxt;
}e[maxn*4];
int cnt;
int head[maxn];
void add_edge(int u,int v)
{
	e[cnt].v=v;
	e[cnt].nxt=head[u];
	head[u]=cnt++;
}
int tim;
int sccnum;
int low[maxn],dfn[maxn];
stack<int> s; 
int scc[maxn];
void tarjan(int u)
{
	low[u]=dfn[u]=++tim;
	s.push(u);
	for(int i=head[u];i!=-1;i=e[i].nxt)
	{
		int v=e[i].v;
		if(!dfn[v]){
			tarjan(v);
			low[u]=min(low[u],low[v]);
		}else if(!scc[v])
		{
			low[u]=min(low[u],dfn[v]);
		}
	}
	if(dfn[u]==low[u])
	{
		sccnum++;
		while(true)
		{
			int x=s.top();s.pop();
			scc[x]=sccnum;
			if(x==u)break;
		}
	}
}
int a[maxn];
int main()
{
	int n,m;
	//freopen("in.txt","r",stdin);
	while(~scanf("%d%d",&n,&m)&&n)
	{
	tim=0;sccnum=0;	cnt=0;
	memset(dfn,0,sizeof dfn);
	memset(low,0,sizeof low);
	memset(scc,0,sizeof scc);
	memset(head,-1,sizeof head);
	while(!s.empty())s.pop();
	int all=0;
	
	for(int i=0;i<n;i++){
		scanf("%d",&a[i]);
		all+=a[i];
	}
	
	for(int i=0;i<m;i++){
		int u,v;
		cin>>u>>v;u--,v--;
		int ok=((a[u]*n)>=all)==((a[v]*n)>=all);
		if(!ok){//不同 
			add_edge(u<<1|1,v<<1);
			add_edge(v<<1|1,u<<1);
		}else {
			add_edge(u<<1,v<<1|1);
			add_edge(v<<1,u<<1|1);
			add_edge(u<<1|1,v<<1);
			add_edge(v<<1|1,u<<1); 
		}
	}
	for(int i=0;i<2*n;i++){
		if(!dfn[i])tarjan(i);
	}
	int ok=0;
	for(int i=0;i<n;i++)
	{
		if(scc[i<<1]==scc[i<<1|1]){
			ok=1;break;
		}
	}
	if(ok){
		cout<<"No solution."<<endl; 
	}
	else {
	for(int i=0;i<n;i++){
		if(scc[i<<1]<scc[i<<1|1]){
			if(a[i]*n>=all){
				puts("A");
			}else {
				puts("B");
			}
		}else {
			puts("C");
		}
	}	
	}	
	}
	
}
