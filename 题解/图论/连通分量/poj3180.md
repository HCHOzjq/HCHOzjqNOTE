```C++
#include<iostream>
#include<cstring>
#include<stack>
using namespace std;
int ans;
const int maxn=100005;
int low[maxn],dfn[maxn];
int scc[maxn];
stack<int> s;
struct edge
{
	int u,v,nxt;
}e[maxn];
int cnt;
int head[maxn];
void add_edge(int u,int v)
{
	e[cnt].u=u;e[cnt].v=v;
	e[cnt].nxt=head[u];
	head[u]=cnt++;
}
int tim;
int sccnum;
void tarjan(int u)
{
	low[u]=dfn[u]=++tim;
	s.push(u);
	for(int i=head[u];i!=-1;i=e[i].nxt)
	{
		int v=e[i].v;
		if(!dfn[v]){
			tarjan(v);
			low[u]=min(low[u],low[v]);
		}else if(!scc[v])
		{
			low[u]=min(low[u],dfn[v]);
		}
	}

	if(dfn[u]==low[u])
	{
		int z=0;
		sccnum++;
		while(true)
		{
			int x=s.top();s.pop();
			scc[x]=sccnum;
			if(x==u)break;
		}
	}
}
int timm[maxn];
int main()
{int n,m;
	while(cin>>n>>m)
	{
		while(!s.empty())s.pop();
		memset(scc,0,sizeof scc);
	memset(head,-1,sizeof head);
	memset(dfn,0,sizeof dfn);
	memset(low,0,sizeof low);
	memset(timm,0,sizeof timm);
	ans=0;
	sccnum=0;
	tim=0;
	cnt=0;
	for(int i=0;i<m;i++)
	{
		int a,b;cin>>a>>b;
		add_edge(a,b);
	}
	for(int i=1;i<=n;i++){
		if(!dfn[i])tarjan(i);
	}
	for(int i=1;i<=n;i++){
		timm[scc[i]]++;
	}
	for(int i=1;i<=sccnum;i++){
		if(timm[i]>=2)ans++;
	}
	cout<<ans<<endl;	
	}
}
```
