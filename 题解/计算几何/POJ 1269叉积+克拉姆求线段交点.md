### POJ 1269

叉积判断是不是在同一条直线上，转换除法为乘法判断是不是平行。

可先判断平行，再判断同一条直线。

要学的是，克拉默法则求交点。

可以推，一般套板子吧。

double要用f输出很迷。

```C++
#include<iostream>
#include<cstring>
#include<cmath>
#include<cstdio>
using namespace std;
struct point{
	double x,y;
	point(double a,double b){
		x=a;
		y=b;
	}	
};
double xmul(point a,point b){
	return b.y*a.x-b.x*a.y;
}

point line(point a,point b){
	return point(b.x-a.x,b.y-a.y);
}

bool ok(point a,point b,point c,point d){
	if(xmul(line(d,c),line(d,a))==0&&xmul(line(d,c),line(d,b))==0){
		return true;
	}else return false;
}

int main()
{
	int n;
	cin>>n;
	cout<<"INTERSECTING LINES OUTPUT"<<endl;
	for(int i=0;i<n;i++){
		double x1,x2,x3,x4;
		double y1,y2,y3,y4;
		
		cin>>x1>>y1>>x2>>y2;
		point a=point(x1,y1);
		point b=point(x2,y2);
		
		cin>>x3>>y3>>x4>>y4;
		point c=point(x3,y3);
		point d=point(x4,y4);
		if((x2-x1)*(y4-y3)==(y2-y1)*(x4-x3)){
			if(ok(a,b,c,d))
			cout<<"LINE"<<endl;
			else cout<<"NONE"<<endl;
		}else {
			double a1=y1-y2,b1=x2-x1,c1=x1*y2-x2*y1;
			double a2=y3-y4,b2=x4-x3,c2=x3*y4-x4*y3;
			double xx=-(c1*b2-b1*c2)/(a1*b2-b1*a2);
			double yy=-(a1*c2-c1*a2)/(a1*b2-b1*a2);
			printf("POINT %.2f %.2f\n",xx,yy);
			
		}
	}
	cout<<"END OF OUTPUT"<<endl;
}
```
