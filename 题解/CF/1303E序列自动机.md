```C++
#include<bits/stdc++.h>

using namespace std;

const int maxn=1004;

char a[maxn],b[maxn];
int dp[maxn][maxn];
int nxt[maxn][maxn];
int n,m;

void get_nxt()
{
	for(int i=0;i<26;i++)nxt[m][i]=m+1;
	
	for(int i=m;i>=1;i--){
		for(int j=0;j<26;j++){
			nxt[i-1][j]=nxt[i][j];
		}
		nxt[i-1][a[i]-'a']=i;
	}
}

bool solve(int pos)
{
	int lena=pos;
	int lenb=n-pos;
	dp[0][0]=0;
	for(int i=0;i<=lena;i++){
		for(int j=0;j<=lenb;j++){
			if(!i&&!j)continue;
			dp[i][j]=m+1;
			if(i>0&&dp[i-1][j]<m)dp[i][j]=min(dp[i][j],nxt[dp[i-1][j]][b[i]-'a']);
			if(j>0&&dp[i][j-1]<m)dp[i][j]=min(dp[i][j],nxt[dp[i][j-1]][b[lena+j]-'a']);
		}
	}
	return dp[lena][lenb]<=m;
}

int main()
{
	int t;cin>>t;
	
	while(t--)
	{
		scanf("%s%s",a+1,b+1);
		m=strlen(a+1);
		n=strlen(b+1);
		get_nxt();
		int ans=0;
		for(int i=1;i<=n;i++)
		{
			if(solve(i))ans=1;
		}
		
		cout<<(ans==1?"YES":"NO")<<endl;
	}
}
```

可以`nxt[i][j]`中i后面j字母的第一个位置。
