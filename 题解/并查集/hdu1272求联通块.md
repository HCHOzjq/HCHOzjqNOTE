求联通块，诶，求互相连接起来，最后看看有没有父节点是一样的。以及所有节点有没有是根节点的，就一定为`No`。

```C++
#include<iostream>
#include<set>
#include<cstdio>
using namespace std;
int fa[100005];
int find(int x)
{
	if(x==fa[x])return x;
	return fa[x]=find(fa[x]);
}
int main()
{
	int p,q;
	
	while(scanf("%d%d",&p,&q))
	{
		if(p==-1&&q==-1)return 0;
		
		if(p==0&&q==0){
			cout<<"Yes"<<endl;continue;
		}
		set<int> s;
		for(int i=0;i<=100000;i++)fa[i]=i;
		int x=find(p),y=find(q);
		fa[x]=y;
		s.insert(p);s.insert(q);
		int ans=1;
		while(scanf("%d%d",&p,&q)){
			if(p==q&&p==0)break;
			
			x=find(p),y=find(q);
			s.insert(p);s.insert(q);	
			if(x==y){
				ans=0;
			}else {
				fa[x]=y;
			}
		}
		int cnt=0;
		for(auto it=s.begin();it!=s.end();it++)
		{
			int tmp=*it;
			if(fa[tmp]==tmp)cnt++;
		}
		if(cnt>1)ans=0;
		printf("%s\n",(ans==1)?"Yes":"No");
	}
}
```
