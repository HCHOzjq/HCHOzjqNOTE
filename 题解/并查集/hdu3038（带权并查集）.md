### HDU 3038

板子题目，注意的是`get`函数也需要将数据加上去。留在这里给个印象。

```C++
#include<bits/stdc++.h>

using namespace std;

int val[200005];
int fa[200005];

int get(int x)
{
	if(x==fa[x])return x;
	int t=fa[x];
	fa[x]=get(fa[x]);
	val[x]+=val[t];
	return fa[x];
}

void init(int n)
{
	for(int i=0;i<=n;i++){
		fa[i]=i;
		val[i]=0;
	}
}


int main()
{
	int n,m;
	while(~scanf("%d%d",&n,&m))
	{
	init(n);
	int fl=0;
	for(int i=1;i<=m;i++)
	{
		int a,b,c;
		scanf("%d%d%d",&a,&b,&c);
		a--;
		int x=get(a),y=get(b);
		if(x!=y){
			fa[y]=x;
			val[y]=val[a]+c-val[b];
		}else {
			if(val[b]!=val[a]+c){
				fl++;
			}
		}
	}
	cout<<fl<<endl;	
	}
	
}
```
